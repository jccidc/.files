```rust
// src-tauri/src/commands/archive.rs

use std::path::Path;
use std::fs;

#[tauri::command]
pub fn extract_archive(path: String, dest: Option<String>) -> Result<String, String> {
    let src = Path::new(&path);
    let ext = src.extension().and_then(|e| e.to_str()).unwrap_or("").to_lowercase();
    let parent = src.parent().unwrap_or(Path::new("."));
    let stem = src.file_stem().and_then(|s| s.to_str()).unwrap_or("extracted");

    let out_dir = match dest {
        Some(d) => std::path::PathBuf::from(d),
        None => parent.join(stem),
    };

    if !out_dir.exists() {
        fs::create_dir_all(&out_dir).map_err(|e| e.to_string())?;
    }

    match ext.as_str() {
        "zip" => extract_zip(&path, &out_dir),
        "7z" => extract_7z(&path, &out_dir),
        "tar" => extract_tar(&path, &out_dir, None),
        "gz" | "tgz" => extract_tar(&path, &out_dir, Some("gz")),
        "bz2" | "tbz2" => extract_tar(&path, &out_dir, Some("bz2")),
        _ => Err(format!("Unsupported archive format: .{}", ext)),
    }?;

    Ok(out_dir.to_string_lossy().to_string())
}

fn extract_zip(path: &str, out_dir: &Path) -> Result<(), String> {
    let file = fs::File::open(path).map_err(|e| e.to_string())?;
    let mut archive = zip::ZipArchive::new(file).map_err(|e| e.to_string())?;
    archive.extract(out_dir).map_err(|e| e.to_string())
}

fn extract_7z(path: &str, out_dir: &Path) -> Result<(), String> {
    sevenz_rust::decompress_file(path, out_dir).map_err(|e| e.to_string())
}

fn extract_tar(path: &str, out_dir: &Path, compression: Option<&str>) -> Result<(), String> {
    let file = fs::File::open(path).map_err(|e| e.to_string())?;
    match compression {
        Some("gz") => {
            let decoder = flate2::read::GzDecoder::new(file);
            let mut archive = tar::Archive::new(decoder);
            archive.unpack(out_dir).map_err(|e| e.to_string())
        }
        Some("bz2") => {
            let decoder = bzip2::read::BzDecoder::new(file);
            let mut archive = tar::Archive::new(decoder);
            archive.unpack(out_dir).map_err(|e| e.to_string())
        }
        _ => {
            let mut archive = tar::Archive::new(file);
            archive.unpack(out_dir).map_err(|e| e.to_string())
        }
    }
}

#[tauri::command]
pub fn compress_archive(paths: Vec<String>, dest: String, format: String) -> Result<String, String> {
    match format.as_str() {
        "zip" => compress_zip(&paths, &dest),
        "7z" => compress_7z(&paths, &dest),
        _ => Err(format!("Unsupported format: {}", format)),
    }?;
    Ok(dest)
}

fn compress_zip(paths: &[String], dest: &str) -> Result<(), String> {
    let file = fs::File::create(dest).map_err(|e| e.to_string())?;
    let mut writer = zip::ZipWriter::new(file);
    let options = zip::write::SimpleFileOptions::default()
        .compression_method(zip::CompressionMethod::Deflated);

    for p in paths {
        let path = Path::new(p);
        if path.is_file() {
            let name = path.file_name().unwrap_or_default().to_string_lossy();
            writer.start_file(name.to_string(), options).map_err(|e| e.to_string())?;
            let data = fs::read(p).map_err(|e| e.to_string())?;
            use std::io::Write;
            writer.write_all(&data).map_err(|e| e.to_string())?;
        } else if path.is_dir() {
            add_dir_to_zip(&mut writer, path, path, options)?;
        }
    }

    writer.finish().map_err(|e| e.to_string())?;
    Ok(())
}

fn add_dir_to_zip(
    writer: &mut zip::ZipWriter<fs::File>,
    base: &Path,
    current: &Path,
    options: zip::write::SimpleFileOptions,
) -> Result<(), String> {
    for entry in fs::read_dir(current).map_err(|e| e.to_string())? {
        let entry = entry.map_err(|e| e.to_string())?;
        let path = entry.path();
        let relative = path.strip_prefix(base).unwrap_or(&path);
        if path.is_file() {
            writer.start_file(relative.to_string_lossy().to_string(), options).map_err(|e| e.to_string())?;
            let data = fs::read(&path).map_err(|e| e.to_string())?;
            use std::io::Write;
            writer.write_all(&data).map_err(|e| e.to_string())?;
        } else if path.is_dir() {
            add_dir_to_zip(writer, base, &path, options)?;
        }
    }
    Ok(())
}

fn compress_7z(paths: &[String], dest: &str) -> Result<(), String> {
    use sevenz_rust::SevenZWriter;
    let dest_path = Path::new(dest);
    let mut writer = SevenZWriter::create(dest_path).map_err(|e| e.to_string())?;
    for p in paths {
        let path = Path::new(p);
        if path.is_file() {
            let name = path.file_name().unwrap_or_default().to_string_lossy().to_string();
            let data = fs::read(p).map_err(|e| e.to_string())?;
            writer.push_archive_entry(
                sevenz_rust::SevenZArchiveEntry::from(name),
                Some(std::io::Cursor::new(data)),
            ).map_err(|e| e.to_string())?;
        }
    }
    writer.finish().map_err(|e| e.to_string())?;
    Ok(())
}

#[tauri::command]
pub fn list_archive(path: String) -> Result<Vec<String>, String> {
    let src = Path::new(&path);
    let ext = src.extension().and_then(|e| e.to_str()).unwrap_or("").to_lowercase();

    match ext.as_str() {
        "zip" => {
            let file = fs::File::open(&path).map_err(|e| e.to_string())?;
            let archive = zip::ZipArchive::new(file).map_err(|e| e.to_string())?;
            Ok(archive.file_names().map(|n| n.to_string()).collect())
        }
        "7z" => {
            // sevenz-rust doesn't have a clean list API; return placeholder
            Ok(vec!["(7z listing not yet supported)".to_string()])
        }
        _ => Err(format!("Unsupported format for listing: .{}", ext)),
    }
}
```
